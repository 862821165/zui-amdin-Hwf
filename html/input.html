    $(".num_add").click(function(){
        var num=$(this).siblings("#num").attr('value');
        num++;
        $(this).siblings("#num").attr('value',num);
    })
    $(".num_reduce").click(function(){
        var num=$(this).siblings("#num").attr('value');
        if(num>1){
            num--;
            $(this).siblings("#num").attr('value',num);
            if(num==1){
                return
            }
        }
    })
